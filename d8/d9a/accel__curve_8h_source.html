249  static float calcReachableVelocityEnd(const float j_max,
250  const float a_max,
251  const float vs,
252  const float vt,
253  const float d) {
254  /* 速度が曲線となる部分の時間を決定 */
255  const auto tc = a_max / j_max;
256  /* 最大加速度の符号を決定 */
257  const auto am = (vt > vs) ? a_max : -a_max;
258  const auto jm = (vt > vs) ? j_max : -j_max;
259  /* 等加速度直線運動の有無で分岐 */
260  const auto d_triangle = (vs + am * tc / 2) * tc; //< distance @ tm == 0
261  const auto v_triangle = jm / am * d - vs; //< v_end @ tm == 0
262  // ctrl_logd << "d_tri: " << d_triangle << std::endl;
263  // ctrl_logd << "v_tri: " << v_triangle << std::endl;
264  if (d * v_triangle > 0 && std::abs(d) > std::abs(d_triangle)) {
265  /* 曲線・直線・曲線 */
266  ctrl_logd << "v: curve - straight - curve" << std::endl;
267  /* 2次方程式の解の公式を解く */
268  const auto amtc = am * tc;
269  const auto D = amtc * amtc - 4 * (amtc * vs - vs * vs - 2 * am * d);
270  const auto sqrtD = std::sqrt(D);
271  return (-amtc + (d > 0 ? sqrtD : -sqrtD)) / 2;
272  }
273  /* 曲線・曲線 (走行距離が短すぎる) */
274  /* 3次方程式を解いて，終点速度を算出;
275  * 簡単のため，値を一度すべて正に変換して，計算結果に符号を付与して返送 */
276  const auto a = std::abs(vs);
277  const auto b = (d > 0 ? 1 : -1) * jm * d * d;
278  const auto aaa_27 = a * a * a / 27;
279  const auto cr = 8 * aaa_27 + b / 2;
280  const auto ci_b = 8 * aaa_27 / b + 1.0f / 4;
281  if (ci_b >= 0) {
282  /* ルートの中が非負のとき，3乗根により解を求める */
283  ctrl_logd << "v: curve - curve (accel)" << std::endl;
284  const auto c = std::cbrt(cr + std::abs(b) * std::sqrt(ci_b));
285  return (d > 0 ? 1 : -1) * (c + 4 * a * a / c / 9 - a / 3);
286  } else {
287  /* ルートの中が負のとき，極座標変換して解を求める */
288  ctrl_logd << "v: curve - curve (decel)" << std::endl;
289  const auto ci = std::abs(b) * std::sqrt(-ci_b);
290  const auto r = std::hypot(cr, ci); //< = sqrt(cr^2 + ci^2)
291  const auto th = std::atan2(ci, cr);
292  return (d > 0 ? 1 : -1) * (2 * std::cbrt(r) * std::cos(th / 3) - a / 3);
293  }
294  }
