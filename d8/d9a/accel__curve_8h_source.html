235  static float calcReachableVelocityEnd(const float j_max, const float a_max,
236  const float vs, const float vt,
237  const float d) {
238  /* 速度が曲線となる部分の時間を決定 */
239  const auto tc = a_max / j_max;
240  /* 最大加速度の符号を決定 */
241  const auto am = (vt > vs) ? a_max : -a_max;
242  const auto jm = (vt > vs) ? j_max : -j_max;
243  /* 等加速度直線運動の有無で分岐 */
244  const auto d_triangle = (vs + am * tc / 2) * tc; //< distance @ tm == 0
245  const auto v_triangle = jm / am * d - vs; //< v_end @ tm == 0
246  // logd << "d_tri: " << d_triangle << std::endl;
247  // logd << "v_tri: " << v_triangle << std::endl;
248  if (d * v_triangle > 0 && std::abs(d) > std::abs(d_triangle)) {
249  /* 曲線・直線・曲線 */
250  logd << "v: curve - straight - curve" << std::endl;
251  /* 2次方程式の解の公式を解く */
252  const auto amtc = am * tc;
253  const auto D = amtc * amtc - 4 * (amtc * vs - vs * vs - 2 * am * d);
254  const auto sqrtD = std::sqrt(D);
255  return (-amtc + (d > 0 ? sqrtD : -sqrtD)) / 2;
256  }
257  /* 曲線・曲線 (走行距離が短すぎる) */
258  /* 3次方程式を解いて，終点速度を算出;
259  * 簡単のため，値を一度すべて正に変換して，計算結果に符号を付与して返送 */
260  const auto a = std::abs(vs);
261  const auto b = (d > 0 ? 1 : -1) * jm * d * d;
262  const auto aaa_27 = a * a * a / 27;
263  const auto cr = 8 * aaa_27 + b / 2;
264  const auto ci_b = 8 * aaa_27 / b + 1.0f / 4;
265  if (ci_b >= 0) {
266  /* ルートの中が非負のとき，3乗根により解を求める */
267  logd << "v: curve - curve (accel)" << std::endl;
268  const auto c = std::cbrt(cr + std::abs(b) * std::sqrt(ci_b));
269  return (d > 0 ? 1 : -1) * (c + 4 * a * a / c / 9 - a / 3);
270  } else {
271  /* ルートの中が負のとき，極座標変換して解を求める */
272  logd << "v: curve - curve (decel)" << std::endl;
273  const auto ci = std::abs(b) * std::sqrt(-ci_b);
274  const auto r = std::hypot(cr, ci); //< = sqrt(cr^2 + ci^2)
275  const auto th = std::atan2(ci, cr);
276  return (d > 0 ? 1 : -1) * (2 * std::cbrt(r) * std::cos(th / 3) - a / 3);
277  }
