113  static void integrate(const AccelDesigner &ad, State &s, const float v,
114  const float t, const float Ts) {
115  /* Calculation */
116  const std::array<float, 3> th{{ad.x(t), ad.x(t + Ts / 2), ad.x(t + Ts)}};
117  std::array<float, 3> cos_th;
118  std::array<float, 3> sin_th;
119  for (int i = 0; i < 3; ++i) {
120  cos_th[i] = std::cos(th[i]);
121  sin_th[i] = std::sin(th[i]);
122  }
123  /* Runge-Kutta Integral */
124  s.q.x += v * Ts * (cos_th[0] + 4 * cos_th[1] + cos_th[2]) / 6;
125  s.q.y += v * Ts * (sin_th[0] + 4 * sin_th[1] + sin_th[2]) / 6;
126  /* Result */
127  s.dq.x = v * cos_th[2];
128  s.dq.y = v * sin_th[2];
129  s.q.th = ad.x(t + Ts);
130  s.dq.th = ad.v(t + Ts);
131  s.ddq.th = ad.a(t + Ts);
132  s.dddq.th = ad.j(t + Ts);
133  s.ddq.x = -s.dq.y * s.dq.th;
134  s.ddq.y = +s.dq.x * s.dq.th;
135  s.dddq.x = -s.ddq.y * s.dq.th - s.dq.y * s.ddq.th;
136  s.dddq.y = +s.ddq.x * s.dq.th + s.dq.x * s.ddq.th;
137  }
