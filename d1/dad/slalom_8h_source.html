141  static void integrate(const AccelDesigner& ad,
142  State& s,
143  const float v,
144  const float t,
145  const float Ts,
146  const float k_slip = 0) {
147  /* Calculation */
148  const std::array<float, 3> th{{ad.x(t), ad.x(t + Ts / 2), ad.x(t + Ts)}};
149  const std::array<float, 3> w{{ad.v(t), ad.v(t + Ts / 2), ad.v(t + Ts)}};
150  std::array<float, 3> cos_th;
151  std::array<float, 3> sin_th;
152  for (int i = 0; i < 3; ++i) {
153  const auto th_slip = std::atan(-k_slip * v * w[i]);
154  cos_th[i] = std::cos(th[i] + th_slip);
155  sin_th[i] = std::sin(th[i] + th_slip);
156  }
157  /* Runge-Kutta Integral */
158  s.q.x += v * Ts * (cos_th[0] + 4 * cos_th[1] + cos_th[2]) / 6;
159  s.q.y += v * Ts * (sin_th[0] + 4 * sin_th[1] + sin_th[2]) / 6;
160  /* Result */
161  s.dq.x = v * cos_th[2];
162  s.dq.y = v * sin_th[2];
163  s.q.th = ad.x(t + Ts);
164  s.dq.th = ad.v(t + Ts);
165  s.ddq.th = ad.a(t + Ts);
166  s.dddq.th = ad.j(t + Ts);
167  s.ddq.x = -s.dq.y * s.dq.th;
168  s.ddq.y = +s.dq.x * s.dq.th;
169  s.dddq.x = -s.ddq.y * s.dq.th - s.dq.y * s.ddq.th;
170  s.dddq.y = +s.ddq.x * s.dq.th + s.dq.x * s.ddq.th;
171  }
